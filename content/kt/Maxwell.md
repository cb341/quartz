\begin{array}{l}  
\text{Definitions} \\

\varepsilon_0 := \text{electric permitivity in vacuum} \\

Equation 0

$$
\rho = \frac{Q}{A}
$$

Equation 1 (Gauss's Law of Magnetism)

$$
\nabla \cdot \vec{B} = 0
$$

$$
\begin{array}{l}  
\text{An outgoing magnetic field is caused by "nothing"} \\  
\implies  
\text{An outgoing magnetic field (Monopole) does not exist in our Universe}  
\end{array}
$$

Equation 2 (Gauss Law for Electricity)

$$
\nabla \times\vec{E} = \frac{\rho}{\varepsilon_0}
$$

$$
\begin{array}{l}
\text{Curl of the electric field is proportional to the charge density and} \\
\text{inversely proportional to the electric permitivity in a vacuum}
\end{array}
$$

Equation 3 (Faraday's Law of induction)

$$
\nabla \cdot \vec{E} = \frac{-d\vec{B}}{dt}
$$

$$
\begin{array}{l}
\text{A magnetic field changing in respect to time induces a change in electric field.} \\
\text{Changes in the magnetic field induce a current on the electric field with respect to time.}
\end{array}
$$


Equation 4 (Ampere's Law)

$$
\nabla\times\vec{B} = \mu_0J + \mu_0\varepsilon_0\frac{d\vec{E}}{dt}
$$

$$
\begin{array}{l}
\text{The diverging magnetic field is stronger with a larger current density} \\
\text{The outgoing magnetic field is also affected by the change of electric field -} \\
\text{with respect to time and permibility / permitivity of free space}
\end{array}
$$
